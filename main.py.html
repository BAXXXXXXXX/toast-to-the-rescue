main.py 
import pygame
import sys
import random
import math

pygame.init()
pygame.mixer.init()

WIDTH, HEIGHT = 900, 500
screen = pygame.display.set_mode((WIDTH, HEIGHT))
pygame.display.set_caption("Toast to the Rescue!")

clock = pygame.time.Clock()

# ---------- COLORS ----------
SKY_DAY_TOP = (135, 206, 250)
SKY_DAY_BOTTOM = (180, 225, 250)
ROAD_COLOR = (50, 50, 50)
BROWN = (210, 140, 60)
RED = (220, 50, 50)
GRAY = (130, 130, 130)
YELLOW = (255, 230, 0)
BLACK = (0, 0, 0)
WHITE = (255, 255, 255)
GREEN = (50, 200, 50)
BLUE = (50, 50, 200)
CAR_COLORS = [RED, GREEN, BLUE, (255,140,0), (255,0,255)]

font = pygame.font.SysFont(None, 32)
big_font = pygame.font.SysFont(None, 72)

# ---------- SOUND ----------
def beep(freq, dur=150):
    sr = 44100
    n = int(sr * dur / 1000)
    buf = bytearray()
    for i in range(n):
        t = i / sr
        buf.append(int(127 * math.sin(2 * math.pi * freq * t)) + 128)
    return pygame.mixer.Sound(buffer=bytes(buf))

s_save = beep(880)
s_hit = beep(220)
s_gameover = beep(110, 300)

# Music
music_notes = [
    beep(330, 200), beep(440, 200), beep(550, 150), beep(660, 250),
    beep(550, 150), beep(440, 200), beep(330, 200), beep(440, 200)
]
music_index = 0
music_channel = pygame.mixer.Channel(0)
music_timer = 0
MUSIC_INTERVAL = 15

# ---------- PLAYER ----------
player = pygame.Rect(100, 360, 40, 40)
vel_y = 0
gravity = 0.9
jump = -16
on_ground = False
invincible = 0

# ---------- CARS ----------
cars = []

# ---------- XYLOPHONES ----------
xylophones = []
XYL_WIDTH, XYL_HEIGHT = 30, 20
XYL_TIME = 300

# ---------- RECORD ----------
record = 0  # punteggio più alto della sessione

# ---------- GAME STATE ----------
def reset():
    global score, lives, game_over, cars, car_speed, invincible, xylophones
    score = 0
    lives = 3
    game_over = False
    car_speed = 3
    invincible = 0
    cars.clear()
    cars.append(spawn_car())
    xylophones.clear()
    spawn_xylophone()

def spawn_car():
    lane_x = random.randint(50, WIDTH-80)
    color = random.choice(CAR_COLORS)
    return {"rect": pygame.Rect(lane_x, -60, 60, 30), "color": color}

def spawn_xylophone():
    lane_y = 420 - 40
    x = random.randint(100, WIDTH - 50)
    xylophones.append({"rect": pygame.Rect(x, lane_y, XYL_WIDTH, XYL_HEIGHT), "timer": XYL_TIME})

reset()
state = "menu"

# ---------- MAIN LOOP ----------
while True:
    # Sfondo sfumato verticale
    for y in range(HEIGHT):
        ratio = y / HEIGHT
        r = int(SKY_DAY_TOP[0]*(1-ratio) + SKY_DAY_BOTTOM[0]*ratio)
        g = int(SKY_DAY_TOP[1]*(1-ratio) + SKY_DAY_BOTTOM[1]*ratio)
        b = int(SKY_DAY_TOP[2]*(1-ratio) + SKY_DAY_BOTTOM[2]*ratio)
        pygame.draw.line(screen, (r,g,b), (0,y), (WIDTH,y))

    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            pygame.quit()
            sys.exit()

    keys = pygame.key.get_pressed()

    # ---------- MENU ----------
    if state == "menu":
        screen.blit(big_font.render("TOAST TO THE RESCUE!", True, BLACK), (150, 150))
        screen.blit(font.render("Premi SPAZIO per iniziare", True, BLACK), (300, 240))
        screen.blit(font.render(f"Record sessione: {record}", True, BLACK), (330, 280))
        if keys[pygame.K_SPACE]:
            state = "game"
        pygame.display.update()
        clock.tick(60)
        continue

    # ---------- GAME OVER ----------
    if 'game_over' in locals() and game_over:
        screen.blit(big_font.render("GAME OVER", True, BLACK), (300, 180))
        screen.blit(font.render(f"Punteggio: {score}", True, BLACK), (350, 260))
        screen.blit(font.render("Premi R per ricominciare", True, BLACK), (300, 300))
        screen.blit(font.render(f"Record sessione: {record}", True, BLACK), (330, 340))
        if keys[pygame.K_r]:
            reset()
        pygame.display.update()
        clock.tick(60)
        continue

    # ---------- PLAYER ----------
    if keys[pygame.K_LEFT]:
        player.x -= 5
    if keys[pygame.K_RIGHT]:
        player.x += 5
    if keys[pygame.K_SPACE] and on_ground:
        vel_y = jump
        on_ground = False

    vel_y += gravity
    player.y += vel_y
    if player.bottom >= 420:
        player.bottom = 420
        vel_y = 0
        on_ground = True

    # ---------- CARS ----------
    for car in cars:
        car["rect"].y += car_speed
        if car["rect"].y > HEIGHT:
            car["rect"].y = -60
            car["rect"].x = random.randint(50, WIDTH-80)
            car["color"] = random.choice(CAR_COLORS)
        if car["rect"].colliderect(player) and invincible == 0:
            lives -= 1
            s_hit.play()
            invincible = 60

    # ---------- XYLOPHONES ----------
    for xylo in xylophones[:]:
        xylo["timer"] -= 1
        pygame.draw.rect(screen, YELLOW, xylo["rect"])
        if xylo["rect"].colliderect(player):
            score += 1
            s_save.play()
            xylophones.remove(xylo)
            spawn_xylophone()
        elif xylo["timer"] <= 0:
            xylophones.remove(xylo)
            spawn_xylophone()

    if invincible > 0:
        invincible -= 1

    # ---------- DIFFICOLTÀ GRADUALE ----------
    car_speed = 3 + score * 0.05
    if score % 3 == 0 and len(cars) < min(5, score // 3 + 1):
        while len(cars) < min(5, score // 3 + 1):
            cars.append(spawn_car())

    # ---------- GAME OVER CHECK ----------
    if lives <= 0:
        if score > record:
            record = score  # aggiorna record sessione
        game_over = True
        s_gameover.play()

    # ---------- ROAD ----------
    pygame.draw.rect(screen, ROAD_COLOR, (0, 420, WIDTH, 80))

    # ---------- PLAYER DRAW ----------
    if invincible % 10 < 5:
        pygame.draw.rect(screen, BROWN, player)
        pygame.draw.rect(screen, BLACK, player, 2)
        pygame.draw.circle(screen, BLACK, (player.x+12, player.y+15), 3)
        pygame.draw.circle(screen, BLACK, (player.x+28, player.y+15), 3)
        pygame.draw.arc(screen, BLACK, (player.x+12, player.y+25, 16, 6), math.pi, 2*math.pi, 2)

    # ---------- DRAW CARS ----------
    for car in cars:
        pygame.draw.rect(screen, car["color"], car["rect"])
        pygame.draw.rect(screen, GRAY, (car["rect"].x+5, car["rect"].y+5, 50, 20))
        pygame.draw.circle(screen, BLACK, (car["rect"].x+15, car["rect"].y+30), 5)
        pygame.draw.circle(screen, BLACK, (car["rect"].x+45, car["rect"].y+30), 5)

    # ---------- UI ----------
    screen.blit(font.render(f"Score: {score}", True, BLACK), (10, 10))
    screen.blit(font.render(f"Vite: {lives}", True, BLACK), (10, 40))
    screen.blit(font.render(f"Record: {record}", True, BLACK), (WIDTH-150, 10))

    for i in range(lives):
        pygame.draw.circle(screen, RED, (20 + i*25, 70), 8)

    # ---------- MUSIC ----------
    music_timer += 1
    if music_timer >= MUSIC_INTERVAL:
        music_timer = 0
        music_channel.play(music_notes[music_index])
        music_index = (music_index + 1) % len(music_notes)

    pygame.display.update()
    clock.tick(60)
